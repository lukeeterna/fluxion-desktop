79. QUERY APPUNTAMENTI CON DETTAGLI:

    sqlite3 ~/Library/Application\ Support/com.fluxion.app/fluxion.db \
      "SELECT
         a.data_ora_inizio,
         c.nome || ' ' || c.cognome as cliente,
         s.nome as servizio,
         o.nome || ' ' || o.cognome as operatore,
         a.prezzo
       FROM appuntamenti a
       JOIN clienti c ON a.cliente_id = c.id
       JOIN servizi s ON a.servizio_id = s.id
       LEFT JOIN operatori o ON a.operatore_id = o.id
       ORDER BY a.data_ora_inizio
       LIMIT 5;"
